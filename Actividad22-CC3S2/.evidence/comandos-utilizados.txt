# Comandos utilizados en la actividad

# Preparar entorno
cd Observabilidad-mcp
python -m venv bdd
source bdd/bin/activate
make deps

# Levantar stack
make up
docker ps

# Verificar servicios

curl http://localhost:3100/ready

# Ver logs de contenedores
docker logs demo-app --tail 20

docker logs loki --tail 20

# Reiniciar si hay problemas
docker restart loki
docker restart promtail
